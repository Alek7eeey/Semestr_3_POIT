int function FindMaxLen(str param x, str param y)
{
	int var result;
	int var lenX; lenX = lenght_str(x);
	int var lenY; lenY = lenght_str(y);
	if(lenX > lenY)
	then
	{
		result = lenX;
	}
	else
	{
		result = lenY;
	}
	return result;
}
